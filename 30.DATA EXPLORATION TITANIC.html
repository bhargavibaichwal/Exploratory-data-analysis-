# fill the missing values in age by the median as per the class
In [13]:
def fill_age(cols):
    Age=cols[0]
    Pclass=cols[1]
    if pd.isnull(Age):
        if Pclass==1:
            return 37
        elif Pclass==2:
            return 28
        else:
            return 24
    else:
        return Age
        
titanic['Age']=titanic[['Age','Pclass']].apply(fill_age,axis=1)
titanic['Age'].isnull().sum()# this shows that there are no null values in the age column
